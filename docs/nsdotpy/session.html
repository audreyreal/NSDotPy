 32def canonicalize(string: str) -> str:

 34
 35    Args:
 36        string (str): The string to convert
 37
 38    Returns:
 39        str: The canonical form of the string
 40    """

142    def _wait_for_input(self, key: str) -> int:
143        """Blocks execution until the user presses a key. Used as the one click = one request action.
144
145        Args:
146            key (str): The key to wait for
147
148        Returns:
149            int: Userclick parameter, milliseconds since the epoch"""
150        keyboard.wait(key)
151        # the trigger_on_release parameter is broken on windows
152        # because of a bug in keyboard so we have to do this
153        while keyboard.is_pressed(key):
154            pass
155        return int(time.time() * 1000)

170    def _validate_fields(self, data: dict):
171        max_lengths = {
172            "pretitle": 28,
173            "slogan": 55,
174            "currency": 40,
175            "animal": 40,
176            "demonym_noun": 44,
177            "demonym_adjective": 44,
178            "demonym_plural": 44,
179        }
180
181        # go through each key in the data dict and make sure they're below the max length
182        for key, value in data.items():
183            if key not in max_lengths:
184                continue
185            if len(value) > max_lengths[key]:
186                raise ValueError(f"{key} is too long, max length is {max_lengths[key]}")
187            if len(value) < 2 and key != "slogan":
188                raise ValueError(f"{key} should have a minimum length of 2 characters.")
189            # check if pretitle contains any non-alphanumeric characters (except spaces)
190            if key == "pretitle" and not value.replace(" ", "").isalnum():
191                raise ValueError(
192                    "Pretitle should only contain alphanumeric characters or spaces."
193                )

195    def _html_request(
196        self, url, data={}, files=None, allow_redirects=False, auth=None
197    ) -> requests.Response:
198        data |= {"chk": self.chk, "localid": self.localid}
199        userclick = self._wait_for_input(self.keybind)
200        # userclick is the number of milliseconds since the epoch, admin uses this for help enforcing the simultaneity rule
201        response = self._session.post(
202            f"{url}/userclick={userclick}",
203            data=data,
204            files=files,
205            allow_redirects=allow_redirects,
206            auth=auth,
207        )
208        if response.status_code >= 400:
209            with open("error.html", "w") as f:
210                f.write(response.text)
211            raise requests.HTTPError(
212                f"Received status code {response.status_code} from {response.url}. Error page saved to error.html."
213            )
214        self._get_auth_values(response)
215        return response

238    def request(
239        self, url: str, data: dict = {}, files: dict = {}, allow_redirects: bool = False
240    ) -> requests.Response:
241        """Sends a request to the given url with the given data and files.
242
243        Args:
244            url (str): URL to send the request to
245            data (dict, optional): Payload to send with the request
246            files (dict, optional): Payload to send with requests that upload files
247
248        Returns:
249            requests.Response: The response from the server
250        """
251        auth = None
252        if self._ns_server != "1":
253            url = url.replace("nationstates.net", f"nationstates{self._ns_server}.net")
254            auth = (self._auth_user, self._auth_password)
255        if any(
256            banned_page in canonicalize(url)
257            for banned_page in [
258                "page=telegrams",
259                "page=dilemmas",
260                "page=compose_telegram",
261                "page=store",
262            ]
263        ):
264            raise ValueError(
265                "You cannot use a tool to interact with telegrams, issues, getting help, or the store. Read up on the script rules: https://forum.nationstates.net/viewtopic.php?p=16394966#p16394966"
266            )
267        if "api.cgi" in canonicalize(url):
268            # deal with ratelimiting if its an api request
269            return self.api_request(data, _auth=auth)
270        elif "nationstates" in canonicalize(url):
271            # do all the things that need to be done for html requests
272            return self._html_request(url, data, files, allow_redirects, auth=auth)
273        else:
274            # if its not nationstates then just pass the request through
275            return self._session.post(url, data=data, allow_redirects=allow_redirects)

277    def api_request(self, data: dict, _auth=None) -> requests.Response:
278        """Sends a request to the nationstates api with the given data.
279
280        Args:
281            data (dict): Payload to send with the request, e.g. {"nation": "testlandia", "q": "region"}
282
283        Returns:
284            requests.Response: The response from the server
285        """
286        # TODO: probably move this responsibility to a third party api library to avoid reinventing the wheel
287        # if one exists of sufficient quality thats AGPLv3 compatible
288        data |= {"v": "12"}
289        url = (
290            f"https://www.nationstates{self._ns_server}.net/cgi-bin/api.cgi"
291            if _auth
292            else "https://www.nationstates.net/cgi-bin/api.cgi"
293        )
294        # rate limiting section
295        response = self._session.post(url, data=data, auth=_auth)
296        # if the server tells us to wait, wait
297        head = response.headers
298        if waiting_time := head.get("Retry-After"):
299            self.logger.warning(f"Rate limited. Waiting {waiting_time} seconds.")
300            time.sleep(int(waiting_time))
301        # slow down requests so we dont hit the rate limit in the first place
302        requests_left = int(head["X-RateLimit-Remaining"])
303        seconds_until_reset = int(head["X-RateLimit-Reset"])
304        time.sleep(seconds_until_reset / requests_left)
305        # end rate limiting section
306        return response

308    def login(self, nation: str, password: str) -> bool:
309        """Logs in to the nationstates site.
310
311        Args:
312            nation (str): Nation name
313            password (str): Nation password
314
315        Returns:
316            bool: True if login was successful, False otherwise
317        """
318        self.logger.info(f"Logging in to {nation}")
319        url = f"https://www.nationstates.net/page=display_region/region={self._auth_region}"
320        # shoutouts to roavin for telling me i had to have page=display_region in the url so it'd work with a userclick parameter
321
322        data = {
323            "nation": canonicalize(nation),
324            "password": password,
325            "theme": "century",
326            "logging_in": "1",
327            "submit": "Login",
328        }
329
330        response = self.request(url, data)
331
332        soup = BeautifulSoup(response.text, "html.parser")
333        # checks if the body tag has your nation name in it; if it does, you're logged in
334        if not soup.find("body", {"data-nname": canonicalize(nation)}):
335            return False
336
337        self.nation = canonicalize(nation)
338        return True

409            "type": pretitle,
410            "slogan": slogan,
411            "currency": currency,
412            "animal": animal,
413            "demonym2": demonym_noun,
414            "demonym": demonym_adjective,

416            "email": email,

423        # make sure everything is following the proper length limits and only contains acceptable characters

279        """Sends a request to the nationstates api with the given data.
280
