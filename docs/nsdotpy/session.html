  38

 139    def _get_auth_values(self, response: httpx.Response):
 140        # make sure it's actually html first
 141        if not response.headers["Content-Type"].startswith("text/html"):
 142            return
 143        # parse the html
 144        soup = BeautifulSoup(response.text, "html.parser")
 145        # gathering chk and localid so i dont have to worry about authenticating l8r
 146        if chk := soup.find("input", {"name": "chk"}):
 147            self.chk = chk["value"].strip()  # type: ignore
 148        if localid := soup.find("input", {"name": "localid"}):
 149            self.localid = localid["value"].strip()  # type: ignore
 150        if pin := self._session.cookies.get("pin"):
 151            # you should never really need the pin but just in case i'll store it
 152            self.pin = pin
 153        if soup.find("a", {"class": "STANDOUT"}):
 154            self.region = canonicalize(
 155                soup.find_all("a", {"class": "STANDOUT"})[1].attrs["href"].split("=")[1]
 156            )

 165    def _wait_for_input(self, key: str) -> int:
 166        """Blocks execution until the user presses a key. Used as the one click = one request action.
 167
 168        Args:
 169            key (str): The key to wait for
 170
 171        Returns:
 172            int: Userclick parameter, milliseconds since the epoch"""
 173        keyboard.wait(key)
 174        # the trigger_on_release parameter is broken on windows
 175        # because of a bug in keyboard so we have to do this
 176        while keyboard.is_pressed(key):
 177            pass
 178        return int(time.time() * 1000)

 193    def _validate_fields(self, data: dict):
 194        max_lengths = {
 195            "pretitle": 28,
 196            "slogan": 55,
 197            "currency": 40,
 198            "animal": 40,
 199            "demonym_noun": 44,
 200            "demonym_adjective": 44,
 201            "demonym_plural": 44,
 202        }
 203
 204        # go through each key in the data dict and make sure they're below the max length
 205        for key, value in data.items():
 206            if key not in max_lengths:
 207                continue
 208            if len(value) > max_lengths[key]:
 209                raise ValueError(f"{key} is too long, max length is {max_lengths[key]}")
 210            if len(value) < 2 and key != "slogan":
 211                raise ValueError(f"{key} should have a minimum length of 2 characters.")
 212            # check if pretitle contains any non-alphanumeric characters (except spaces)
 213            if key == "pretitle" and not value.replace(" ", "").isalnum():
 214                raise ValueError(
 215                    "Pretitle should only contain alphanumeric characters or spaces."
 216                )

 218    def _html_request(
 219        self, url, data={}, files=None, follow_redirects=False, auth=()
 220    ) -> httpx.Response:
 221        data |= {"chk": self.chk, "localid": self.localid}
 222        userclick = self._wait_for_input(self.keybind)
 223        # userclick is the number of milliseconds since the epoch, admin uses this for help enforcing the simultaneity rule
 224        response = self._session.post(
 225            f"{url}/userclick={userclick}",
 226            data=data,
 227            files=files,
 228            follow_redirects=follow_redirects,
 229            auth=auth,
 230        )
 231        if response.status_code >= 400:
 232            with open("error.html", "w") as f:
 233                f.write(response.text)
 234            raise httpx.HTTPError(
 235                f"Received status code {response.status_code} from {response.url}. Error page saved to error.html."
 236            )
 237        self._get_auth_values(response)
 238        return response

 261    def request(
 262        self,
 263        url: str,
 264        data: dict = {},
 265        files: dict = {},
 266        follow_redirects: bool = False,
 267    ) -> httpx.Response:
 268        """Sends a request to the given url with the given data and files.
 269
 270        Args:
 271            url (str): URL to send the request to
 272            data (dict, optional): Payload to send with the request
 273            files (dict, optional): Payload to send with requests that upload files
 274
 275        Returns:
 276            httpx.Response: The response from the server
 277        """
 278        auth = None
 279        if self._ns_server != "1":
 280            url = url.replace("nationstates.net", f"nationstates{self._ns_server}.net")
 281            auth = (self._auth_user, self._auth_password)
 282        if any(
 283            banned_page in canonicalize(url)
 284            for banned_page in [
 285                "page=telegrams",
 286                "page=dilemmas",
 287                "page=compose_telegram",
 288                "page=store",
 289            ]
 290        ):
 291            raise ValueError(
 292                "You cannot use a tool to interact with telegrams, issues, getting help, or the store. Read up on the script rules: https://forum.nationstates.net/viewtopic.php?p=16394966#p16394966"
 293            )
 294        if self._lock:
 295            # if lock is true then we're already in the middle of a
 296            # request and we're in danger of breaking the simultaneity rule
 297            # so raise an error
 298            raise PermissionError(
 299                "You're already in the middle of a request. Stop trying to violate simultaneity."
 300            )
 301        self._lock = True
 302        if "api.cgi" in canonicalize(url):
 303            # deal with ratelimiting if its an api request
 304            response = self.api_request(data, _auth=auth)
 305        elif "nationstates" in canonicalize(url):
 306            # do all the things that need to be done for html requests
 307            response = self._html_request(url, data, files, follow_redirects, auth=auth)
 308        else:
 309            # if its not nationstates then just pass the request through
 310            response = self._session.post(
 311                url, data=data, follow_redirects=follow_redirects
 312            )
 313        self.current_page = (url, response.text)
 314        self._lock = False
 315        return response

 317    def api_request(self, data: dict, _auth=()) -> httpx.Response:
 318        """Sends a request to the nationstates api with the given data.
 319
 320        Args:
 321            data (dict): Payload to send with the request, e.g. {"nation": "testlandia", "q": "region"}
 322
 323        Returns:
 324            httpx.Response: The response from the server
 325        """
 326        # TODO: probably move this responsibility to a third party api library to avoid reinventing the wheel
 327        # if one exists of sufficient quality thats AGPLv3 compatible
 328        data |= {"v": "12"}
 329        url = (
 330            f"https://www.nationstates{self._ns_server}.net/cgi-bin/api.cgi"
 331            if _auth
 332            else "https://www.nationstates.net/cgi-bin/api.cgi"
 333        )
 334        # rate limiting section
 335        response = self._session.post(url, data=data, auth=_auth)
 336        # if the server tells us to wait, wait
 337        head = response.headers
 338        if waiting_time := head.get("Retry-After"):
 339            self.logger.warning(f"Rate limited. Waiting {waiting_time} seconds.")
 340            time.sleep(int(waiting_time))
 341        # slow down requests so we dont hit the rate limit in the first place
 342        requests_left = int(head["X-RateLimit-Remaining"])
 343        seconds_until_reset = int(head["X-RateLimit-Reset"])
 344        time.sleep(seconds_until_reset / requests_left)
 345        # end rate limiting section
 346        return response

 348    def login(self, nation: str, password: str) -> bool:
 349        """Logs in to the nationstates site.
 350
 351        Args:
 352            nation (str): Nation name
 353            password (str): Nation password
 354
 355        Returns:
 356            bool: True if login was successful, False otherwise
 357        """
 358        self.logger.info(f"Logging in to {nation}")
 359        url = f"https://www.nationstates.net/page=display_region/region={self._auth_region}"
 360        # shoutouts to roavin for telling me i had to have page=display_region in the url so it'd work with a userclick parameter
 361
 362        data = {
 363            "nation": canonicalize(nation),
 364            "password": password,
 365            "theme": "century",
 366            "logging_in": "1",
 367            "submit": "Login",
 368        }
 369
 370        response = self.request(url, data)
 371
 372        soup = BeautifulSoup(response.text, "html.parser")
 373        # checks if the body tag has your nation name in it; if it does, you're logged in
 374        if not soup.find("body", {"data-nname": canonicalize(nation)}):
 375            return False
 376
 377        self.nation = canonicalize(nation)
 378        return True

 415    def change_nation_settings(
 416        self,
 417        *,
 418        email: str = "",
 419        pretitle: str = "",
 420        slogan: str = "",
 421        currency: str = "",
 422        animal: str = "",
 423        demonym_noun: str = "",

 425        demonym_plural: str = "",

 427    ) -> bool:
 428        """Given a logged in session, changes customizable fields and settings of the logged in nation.
 429        Variables must be explicitly named in the call to the function, e.g. "session.change_nation_settings(pretitle='Join Lily', currency='Join Lily')"
 430
 431        Args:
 432            email (str, optional): New email for WA apps.
 433            pretitle (str, optional): New pretitle of the nation. Max length of 28.
 434            slogan (str, optional): New Slogan/Motto of the nation. Max length of 55.
 435            currency (str, optional): New currency of the nation. Max length of 40.
 436            animal (str, optional): New national animal of the nation. Max length of 40.
 437            demonym_noun (str, optional): Noun the nation will refer to its citizens as. Max length of 44.
 438            demonym_adjective (str, optional): Adjective the nation will refer to its citizens as. Max length of 44.
 439            demonym_plural (str, optional): Plural form of "demonym_noun". Max length of 44.
 440            new_password (str, optional): New password to assign to the nation.
 441
 442        Returns:
 443            bool: True if changes were successful, False otherwise.
 444        """
 445        self.logger.info(f"Changing settings on {self.nation}")
 446        url = "https://www.nationstates.net/template-overall=none/page=settings"
 447
 448        data = {
 449            "type": pretitle,
 450            "slogan": slogan,
 451            "currency": currency,
 452            "animal": animal,
 453            "demonym2": demonym_noun,
 454            "demonym": demonym_adjective,
 455            "demonym2pl": demonym_plural,
 456            "email": email,
 457            "password": new_password,
 458            "confirm_password": new_password,
 459            "update": " Update ",
 460        }
 461        # remove keys that have empty values
 462        data = {k: v for k, v in data.items() if v}
 463        # make sure everything is following the proper length limits and only contains acceptable characters
 464        self._validate_fields(data)
 465
 466        response = self.request(url, data)

 194    def _validate_fields(self, data: dict):
 195        max_lengths = {
 196            "pretitle": 28,
 197            "slogan": 55,
 198            "currency": 40,
 199            "animal": 40,
 200            "demonym_noun": 44,
 201            "demonym_adjective": 44,
 202            "demonym_plural": 44,
 203        }
 204
 205        # go through each key in the data dict and make sure they're below the max length
 206        for key, value in data.items():
 207            if key not in max_lengths:
 208                continue
 209            if len(value) > max_lengths[key]:
 210                raise ValueError(f"{key} is too long, max length is {max_lengths[key]}")
 211            if len(value) < 2 and key != "slogan":
 212                raise ValueError(f"{key} should have a minimum length of 2 characters.")
 213            # check if pretitle contains any non-alphanumeric characters (except spaces)
 214            if key == "pretitle" and not value.replace(" ", "").isalnum():
 215                raise ValueError(
 216                    "Pretitle should only contain alphanumeric characters or spaces."

 262    def request(
 263        self,
 264        url: str,
 265        data: dict = {},
 266        files: dict = {},
 267        follow_redirects: bool = False,
 268    ) -> httpx.Response:
 269        """Sends a request to the given url with the given data and files.
 270
 271        Args:
 272            url (str): URL to send the request to
 273            data (dict, optional): Payload to send with the request
 274            files (dict, optional): Payload to send with requests that upload files
 275
 276        Returns:
 277            httpx.Response: The response from the server
 278        """
 279        auth = None
 280        if self._ns_server != "1":
 281            url = url.replace("nationstates.net", f"nationstates{self._ns_server}.net")
 282            auth = (self._auth_user, self._auth_password)
 283        if any(
 284            banned_page in canonicalize(url)
 285            for banned_page in [
 286                "page=telegrams",
 287                "page=dilemmas",
 288                "page=compose_telegram",
 289                "page=store",
 290            ]
 291        ):
 292            raise ValueError(
 293                "You cannot use a tool to interact with telegrams, issues, getting help, or the store. Read up on the script rules: https://forum.nationstates.net/viewtopic.php?p=16394966#p16394966"
 294            )
 295        if self._lock:
 296            # if lock is true then we're already in the middle of a
 297            # request and we're in danger of breaking the simultaneity rule
 298            # so raise an error
 299            raise PermissionError(
 300                "You're already in the middle of a request. Stop trying to violate simultaneity."
 301            )
 302        self._lock = True
 303        if "api.cgi" in canonicalize(url):
 304            # deal with ratelimiting if its an api request
 305            response = self.api_request(data, _auth=auth)
 306        elif "nationstates" in canonicalize(url):
 307            # do all the things that need to be done for html requests
 308            response = self._html_request(url, data, files, follow_redirects, auth=auth)
 309        else:
 310            # if its not nationstates then just pass the request through
 311            response = self._session.post(
 312                url, data=data, follow_redirects=follow_redirects
 313            )
 314        self.current_page = (url, response.text)
 315        self._lock = False
 316        return response

 318    def api_request(self, data: dict, _auth=()) -> httpx.Response:
 319        """Sends a request to the nationstates api with the given data.
 320
 321        Args:
 322            data (dict): Payload to send with the request, e.g. {"nation": "testlandia", "q": "region"}
 323
 324        Returns:
 325            httpx.Response: The response from the server
 326        """
 327        # TODO: probably move this responsibility to a third party api library to avoid reinventing the wheel
 328        # if one exists of sufficient quality thats AGPLv3 compatible
 329        data |= {"v": "12"}
 330        url = (
 331            f"https://www.nationstates{self._ns_server}.net/cgi-bin/api.cgi"
 332            if _auth
 333            else "https://www.nationstates.net/cgi-bin/api.cgi"
 334        )
 335        # rate limiting section
 336        response = self._session.post(url, data=data, auth=_auth)
 337        # if the server tells us to wait, wait
 338        head = response.headers
 339        if waiting_time := head.get("Retry-After"):
 340            self.logger.warning(f"Rate limited. Waiting {waiting_time} seconds.")
 341            time.sleep(int(waiting_time))
 342        # slow down requests so we dont hit the rate limit in the first place
 343        requests_left = int(head["X-RateLimit-Remaining"])
 344        seconds_until_reset = int(head["X-RateLimit-Reset"])
 345        time.sleep(seconds_until_reset / requests_left)
 346        # end rate limiting section
 347        return response
